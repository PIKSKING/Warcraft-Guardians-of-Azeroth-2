age_in_relation_to_humans_more_than_trigger = {
	save_temporary_scope_value_as = {
		name = age
		value = {
			value = $age$
			if = {
				limit = { $age$ > 20 }
				multiply = racial_age_multiplier_value
			}
		}
	}
	
	age > scope:age
}
