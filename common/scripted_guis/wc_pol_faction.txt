open_alliance_rename = {
	scope = character
	is_shown = {
		leads_pol_faction_trigger = { faction = alliance }
	}
	is_valid = {
		NOT = { has_variable = gui_union_rename_open }
	}
	effect = {
		set_variable = { name = gui_union_rename_open value = flag:true }
		trigger_event = WCPLF.330
	}
}
