# Warcraft

test_close_family =
{
	expect = {
		any_living_character = {
			count = all
			save_temporary_scope_as = original_character
	
			any_close_family_member = {
				count = all
				is_close_family_of = scope:original_character
				
				any_close_family_member = {
					this = scope:original_character
				}
			}
		}
	}
}
